var g_txtHtml_1 = " &lt;Rufnummer&gt;</td><td>Anrufen über ";
function ProviderDisplay (name, registrar) {
	var provider = registrar;
	switch (registrar) {
		case "freenet.de": 	provider = "Freenet"; break;
		case "sipgate.de": 	provider = "Sipgate"; break;
		case "sip.web.de": 	provider = "WEB.DE"; break;
		case "1und1.de": 		provider = "1&amp;1 Internet"; break;
		case "gmx.de": 		provider = "GMX"; break;
		case "schlund.de": 	provider = "Schlund + Partner"; break;
		case "reg.aolphone.aol.de":	provider = "AOL"; break;
		case "aolphone.aol.de":	provider = "AOL"; break;
		case "strato-iphone.de":	provider = "Strato"; break;
		case "tel.t-online.de":	provider = "T-Online"; break;
	}
	if (registrar.indexOf("sip.arcor.de") >= 0) provider = "Arcor";
	return name+" bei "+provider;
}
function AuswahlDisplay (nr) {
	if (nr == "9") return "*120#";
	return "*12"+(Number(nr)+1)+"#";
}
function CodeDisplay (code, nr) {
	if (nr == "") return "";
	return "**"+code+nr;
}
function SipDisplay (name, registrar, ID) {
	if (name == "") return "";
	return "<tr><td>"+AuswahlDisplay(ID)+g_txtHtml_1+ProviderDisplay(name, registrar)+"</td></tr>";
}
function isVIP (name) {
	return name.length > 0 && name.charAt(0) == '!';
}
function GetBuchName (name) {
	if (isVIP(name)) return name.substr(1);
	return name;
}
function GetVIP (name) {
	if (isVIP(name)) return "X";
	return "";
}
function TrFon(name, rufnr, code, vanity) {
	var str = "<tr>";
	str += "<td>"+GetBuchName(name)+"</td>";
	str += "<td>"+rufnr+"</td>";
	str += "<td>"+CodeDisplay("7", code)+"</td>";
	str += "<td>"+CodeDisplay("8", vanity)+"</td>";
	str += "<td class=\"ac\">"+GetVIP(name)+"</td>";
	str += "</tr>";
	return str;
}
